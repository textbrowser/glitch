58  bool operator<(const QTreeWidgetItem &other) const
59  {
60  auto i = treeWidget()->sortColumn();
61 
62  switch(static_cast<glitch_find_objects::Columns> (i))
63  {
64  case glitch_find_objects::Columns::Object:
65  case glitch_find_objects::Columns::Type:
66  {
67  return other.text(i) > text(i);
68  }
69  case glitch_find_objects::Columns::TypeTotal:
70  {
71  return other.text(i).toInt() > text(i).toInt();
72  }
73  default:
74  {
75  auto list1
76  (other.text(i).remove(' ').remove('(').remove(')').split(','));
77  auto list2(text(i).remove(' ').remove('(').remove(')').split(','));
78  auto x1 = list1.value(0).toInt();
79  auto x2 = list2.value(0).toInt();
80  auto y1 = list1.value(1).toInt();
81  auto y2 = list2.value(1).toInt();
82 
83  return !(x1 < x2 || (x1 == x2 && y1 < y2));
84  }
85  }
86  }
