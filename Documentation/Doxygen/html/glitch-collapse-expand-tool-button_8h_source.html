28#ifndef _glitch_collapse_expand_tool_button_h_
29#define _glitch_collapse_expand_tool_button_h_
30
31#include <QHeaderView>
32#include <QPointer>
33#include <QScrollBar>
34#include <QToolButton>
35#include <QTreeWidget>
36#include <QtDebug>
37
38class glitch_collapse_expand_tool_button: public QToolButton
39{
40 Q_OBJECT
41
42 public:
43 glitch_collapse_expand_tool_button(QTreeWidget *tree):QToolButton(tree)
44 {
45 m_tree = tree;
46
47 if(m_tree)
48 {
49 m_tree->header()->setDefaultAlignment(Qt::AlignCenter);
50 m_tree->header()->setMinimumHeight(30);
51 move(5, (m_tree->header()->size().height() - 25) / 2 + 2);
52 resize(25, 25);
53 setCheckable(true);
54
55 auto font(this->font());
56
57 font.setStyleHint(QFont::Courier);
58 setFont(font);
59 setStyleSheet("QToolButton {border: none;}"
60 "QToolButton::menu-button {border: none;}");
61 setText(tr("+"));
62 setToolTip(tr("Collapse / Expand"));
63 connect(m_tree,
64 SIGNAL(itemCollapsed(QTreeWidgetItem *)),
65 this,
66 SLOT(slotItemCollapsedExpanded(void)));
67 connect(m_tree,
68 SIGNAL(itemExpanded(QTreeWidgetItem *)),
69 this,
70 SLOT(slotItemCollapsedExpanded(void)));
71 connect(m_tree->horizontalScrollBar(),
72 SIGNAL(valueChanged(int)),
73 this,
74 SLOT(slotScroll(int)));
75 connect(this,
76 &glitch_collapse_expand_tool_button::clicked,
77 this,
78 &glitch_collapse_expand_tool_button::slotCollapse);
79 }
80 else
81 {
82 qDebug() << "glitch_collapse_expand_tool_button::"
83 "glitch_collapse_expand_tool_button(): m_tree is zero!";
84 setVisible(false);
85 }
86 }
87
88 private:
89 QPointer<QTreeWidget> m_tree;
90
91 private slots:
92 void slotCollapse(void)
93 {
94 if(m_tree)
95 {
96 disconnect(m_tree,
97 SIGNAL(itemCollapsed(QTreeWidgetItem *)),
98 this,
99 SLOT(slotItemCollapsedExpanded(void)));
100 disconnect(m_tree,
101 SIGNAL(itemExpanded(QTreeWidgetItem *)),
102 this,
103 SLOT(slotItemCollapsedExpanded(void)));
104 }
105
106 if(isChecked())
107 {
108 if(m_tree)
109 m_tree->expandAll();
110
111 setText(tr("-"));
112 }
113 else
114 {
115 if(m_tree)
116 m_tree->collapseAll();
117
118 setText(tr("+"));
119 }
120
121 if(m_tree)
122 {
123 connect(m_tree,
124 SIGNAL(itemCollapsed(QTreeWidgetItem *)),
125 this,
126 SLOT(slotItemCollapsedExpanded(void)));
127 connect(m_tree,
128 SIGNAL(itemExpanded(QTreeWidgetItem *)),
129 this,
130 SLOT(slotItemCollapsedExpanded(void)));
131 }
132 }
133
134 void slotItemCollapsedExpanded(void)
135 {
136 if(!m_tree)
137 return;
138
139 auto expanded = 0;
140
141 for(int i = 0; i < m_tree->topLevelItemCount(); i++)
142 if(m_tree->topLevelItem(i) && m_tree->topLevelItem(i)->childCount() > 0)
143 expanded += m_tree->topLevelItem(i)->isExpanded() ? 1 : 0;
144
145 if(expanded == 0)
146 {
147 setChecked(false);
148 setText(tr("+"));
149 }
150 else
151 {
152 setChecked(true);
153 setText(tr("-"));
154 }
155 }
156
157 void slotScroll(int value)
158 {
159 setVisible(geometry().right() - geometry().width() / 2 > value);
160 }
161};
