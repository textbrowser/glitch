134  static Type stringToCharactersType(const QString &s)
135  {
136  auto string(s.toLower());
137 
138  if(string.contains("isalphanumeric"))
139  return Type::IS_ALPHA_NUMERIC;
140  else if(string.contains("isascii"))
141  return Type::IS_ASCII;
142  else if(string.contains("iscontrol"))
143  return Type::IS_CONTROL;
144  else if(string.contains("isdigit"))
145  return Type::IS_DIGIT;
146  else if(string.contains("isgraph"))
147  return Type::IS_GRAPH;
148  else if(string.contains("ishexadecimaldigit"))
149  return Type::IS_HEXADECIMAL_DIGIT;
150  else if(string.contains("islowercase"))
151  return Type::IS_LOWER_CASE;
152  else if(string.contains("isprintable"))
153  return Type::IS_PRINTABLE;
154  else if(string.contains("ispunct"))
155  return Type::IS_PUNCT;
156  else if(string.contains("isspace"))
157  return Type::IS_SPACE;
158  else if(string.contains("isuppercase"))
159  return Type::IS_UPPER_CASE;
160  else if(string.contains("iswhitespace"))
161  return Type::IS_WHITESPACE;
162  else
163  return Type::IS_ALPHA;
164  }
